# 4. 递归实现

~ function () {
      function myFlat() {
        let newArr = []
        let fn = (arr) => {
          for(let i = 0; i<arr.length;i++) {
            if (Array.isArray(arr[i])){
              fn(arr[i])
              continue;
            }
            newArr.push(arr[i])
          }
        }
        fn(this)
        return newArr
      }
      Array.prototype.myFlat = myFlat
  }()
  console.log(arr.myFlat())

// 二：
function flat(arr) {
    let newArr = []
    for(let i = 0; i<arr.length;i++){
       if(Array.isArray(arr[i])){
           newArr.push(...flat(arr[i]))
          continue
       }
       newArr.push(arr[i])
    }
    return newArr
}
const oldArr = [1, 2, [3, 4]]; 
const newArr = flat(oldArr); 
console.log(newArr); // [1, 2, 3, 4]
